Js代码压缩¶

  • UglifyJs

  • terser webpack4默认使用 terser-webpack-plugin

module.exports = {
    // ....
    optimization: {
        minimize: true
    }
}

// 覆盖默认配置
const TerserPlugin = require('terser-webpack-plugin');
module.exports = {
    // ....
    optimization: {
        minimizer: [
            new TerserPlugin({
                test: /\.js(?.*)?$/i
            })
        ]
    }
}
